    def ustawienie_numerow(self):
        ilosc_liczb=self.ilosc_hexow//2
        liczby=[]
        # dodanie pierwszej polowy
        for i in range(ilosc_liczb):
            liczby.append(4 - i % 5 + 2)

        # utworzenie drugiej polowy
        temp = []
        for i in liczby:
            temp.append(14 - i)
        liczby = liczby + temp

        # tasowanie
        shuffle(liczby)

        # przypisanie do hexow
        for i in range(len(self.plansza)):
            for j in range(len(self.plansza[i])):
                if self.plansza[i][j].surowiec != 5:
                    self.plansza[i][j].dodaj_liczbe(liczby.pop())

    def losowanie_hexow(self):
        surowce=[0,0,0,0,0,1]
        for i in range(self.ilosc_hexow-1):
            surowce[i % 5] += 1

        for i in range(len(self.plansza)):
            for j in range(len(self.plansza[i])):
                wybor=wybor_poczatkowy(surowce)
                #usuwanie sasiadow

                #lewy gorny
                if i > 0 and j > 0 and self.plansza[i - 1][j - 1].surowiec in wybor:
                    wybor.remove(self.plansza[i - 1][j - 1].surowiec)

                # prawy gorny
                if i > 0 and j<len(self.plansza[i-1]) and self.plansza[i - 1][j].surowiec in wybor:
                    wybor.remove(self.plansza[i - 1][j].surowiec)

                # lewy
                if j > 0 and self.plansza[i][j - 1].surowiec in wybor:
                    wybor.remove(self.plansza[i][j - 1].surowiec)

                # prawy
                if j < len(self.plansza[i]) - 1 and self.plansza[i][j + 1].surowiec in wybor:
                    wybor.remove(self.plansza[i][j + 1].surowiec)

                # lewy dolny
                if i < len(self.plansza) - 1 and j>0 and self.plansza[i + 1][j-1].surowiec in wybor:
                    wybor.remove(self.plansza[i + 1][j - 1].surowiec)

                # prawy dolny
                if i < len(self.plansza) - 1 and j<len(self.plansza[i+1]) and self.plansza[i + 1][j].surowiec in wybor:
                    wybor.remove(self.plansza[i + 1][j].surowiec)

                #ustawienie
                if len(wybor)==0:
                    wybor=wybor_poczatkowy(surowce)
                ustawienie=choice(wybor)
                surowce[ustawienie]-=1
                ustaw(self.plansza[i][j],ustawienie)
